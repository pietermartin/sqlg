 public class TestTinkerpopBug extends BaseTest {
 
     @Test(expected = IllegalStateException.class)
     public void hasNextCountBug() {
         this.sqlgGraph.addVertex(T.label, "Person");

         this.sqlgGraph.addVertex(T.label, "Person");

     }

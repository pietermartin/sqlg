         EdgeLabel edgeLabel = this.sqlgGraph.getTopology().getEdgeLabel("A", "test").orElseThrow(IllegalStateException::new);

         } catch (Exception e) {
             //swallow
         }
         this.sqlgGraph.tx().rollback();
